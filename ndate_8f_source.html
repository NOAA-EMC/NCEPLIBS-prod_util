131  nd=log10(jdat(1)+0.5)+1
132  WRITE(cfmt,'("(I",I2,",3I2.2)")') nd
133  print cfmt,jdat(1),jdat(2),jdat(3),jdat(5)
134 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
135  CONTAINS
